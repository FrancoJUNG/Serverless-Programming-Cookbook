aws s3api create-bucket \
    --bucket qnatime \
    --create-bucket-configuration LocationConstraint=ap-south-1 \
    --region ap-south-1 \
    --profile admin

aws s3 cp resources/index.html s3://qnatime/index.html \
    --profile admin

aws s3 cp resources/error.html s3://qnatime/error.html \
    --profile admin

aws s3api put-bucket-website \
    --bucket qnatime \
    --website-configuration file://resources/s3-website-configuration.json \
    --profile admin

aws s3api put-bucket-policy \
    --bucket qnatime \
    --policy file://resources/s3-website-policy.json \
    --profile admin



-- Bucket Website URL --

http://qnatime.s3-website.ap-south-1.amazonaws.com/



-- Cleanup --
aws s3 rb s3://qnatime \
    --force \
    --profile admin



-- Upload files for the CloudFormation template --

aws s3 cp resources/index.html s3://quizzercloud/index.html \
    --profile admin

aws s3 cp resources/error.html s3://quizzercloud/error.html \
    --profile admin
